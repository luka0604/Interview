import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Scanner;

class Scratch {
    public static void main(String[] args) {

        String tekst;
        String tekstNaopako = "";
        Scanner scanner = new Scanner(System.in);
        System.out.println("Upišite tekst kojeg želite provjeriti da li je palindrom : ");
        tekst = scanner.nextLine();
        int length = tekst.length();
        for ( int i = length - 1; i >= 0; i-- )
            tekstNaopako = tekstNaopako + tekst.charAt(i);
        if (tekst.equals(tekstNaopako)) {
            System.out.println("Tekst je palindorm ");
        }
        else {
            System.out.println("Tekst nije palindrom");
        }


    }
}
