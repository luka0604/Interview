import java.sql.SQLOutput;
import java.util.*;

class Scratch {
    public static void main(String[] args) {
        int igrac = 0;
        int računalo = 0;
        Scanner scanner = new Scanner(System.in);
        Random rand = new Random();
        String potez;
        System.out.println("Igra kamen, škare, papir. Za završetak upišite 'kraj'.");
        System.out.println();
        do{
            System.out.println(" Papir, škare ili kamen : ");
            potez = scanner.next();
            int broj = rand.nextInt((3) + 1);
            
            // 1 = kamen, 2 = škare, 3 = papir
            switch (broj) {
                case 1:
                    System.out.println(" Računalo : kamen ");
                    if (potez.equals("kamen") || potez.equals("Kamen")) {
                        System.out.println("ISTO");
                        break;
                    }
                    if (potez.equals("škare") || potez.equals("Škare")) {
                        računalo++;
                        System.out.println("PORAZ");
                        break;
                    }
                    if (potez.equals("papir") || potez.equals("Papir")) {
                        igrac++;
                        System.out.println("POBJEDA");
                        break;
                    }
                case 2:
                    System.out.println(" Računalo : škare " );
                    if (potez.equals("kamen") || potez.equals("Kamen")) {
                        igrac++;
                        System.out.println("POBJEDA");
                        break;
                    }
                    if (potez.equals("škare") || potez.equals("Škare")) {
                        System.out.println("ISTO");
                        break;
                    }
                    if (potez.equals("papir") || potez.equals("Papir")) {
                        računalo++;
                        System.out.println("PORAZ");
                        break;
                    }
                case 3:
                    System.out.println("Računalo : papir ");
                    if (potez.equals("kamen") || potez.equals("Kamen")) {
                        računalo++;
                        System.out.println("PORAZ");
                        break;
                    }
                    if (potez.equals("škare") || potez.equals("Škare")) {
                        igrac++;
                        System.out.println("POBJEDA");
                        break;
                    }
                    if (potez.equals("papir") || potez.equals("Papir")) {
                        System.out.println("ISTO");
                        break;
                    }
                }
            System.out.println();
        } while(!potez.equals("kraj"));
        System.out.println("Računalo : " + računalo);
        System.out.println("Igrač : " + igrac);

    }
}
